55template <typename T, bool detailed> typename allocator_base<T, detailed>::pointer allocator_base<T, detailed>::
56allocate(size_type n_elements) {
57 QALLOC_ASSERT(n_elements > 0);
58 QALLOC_IF_CONSTEXPR(detailed) {
59 return reinterpret_cast<pointer>(m_pool_ptr->template detailed_allocate<T>(n_elements * sizeof(T)));
60 }
61 return reinterpret_cast<pointer>(m_pool_ptr->allocate(n_elements * sizeof(T)));
62}
63
64template <typename T, bool detailed> void allocator_base<T, detailed>::
65deallocate(pointer p, size_type n_elements) {
66 QALLOC_ASSERT(n_elements > 0);
67 if (p == nullptr) return;
68 QALLOC_IF_CONSTEXPR(detailed) {
69 m_pool_ptr->template detailed_deallocate<T>(qalloc::pointer::launder(reinterpret_cast<byte_pointer>(p)), n_elements * sizeof(T));
70 }
71 else {
72 m_pool_ptr->deallocate(qalloc::pointer::launder(reinterpret_cast<byte_pointer>(p)), n_elements * sizeof(T));
73 }
74}
