33template <class T>
34byte_ptr_t pool_t::detailed_allocate(size_type n_bytes_requested) const {
35 byte_ptr_t ptr = pool_t::allocate(n_bytes_requested + sizeof(block_info_t));
36 new (pointer::launder(ptr)) block_info_t{&typeid(T), index_t(m_subpools.size() - 1)};
37 return ptr + sizeof(block_info_t);
38}

41void pool_t::detailed_deallocate(byte_ptr_t p, size_type n_bytes_requested) const {
42 QALLOC_DEBUG_STATEMENT(
43 auto& block = *block_info_t::of(p);
44 QALLOC_ASSERT(size_cast(block.subpool_index) < m_subpools.size());
45 QALLOC_ASSERT(*block.type_info == typeid(T));
46 )
47 pool_t::deallocate(pointer::launder(p - sizeof(block_info_t)), n_bytes_requested + sizeof(block_info_t));

50QALLOC_MAYBE_UNUSED
51size_type pool_t::gc() const { // TODO: fix this, not gc triggers
52 size_type memory_freed = 0;
53 for (auto it = m_freed_blocks.begin(); it != m_freed_blocks.end();) {
54 auto& block = *it;
55 block_info_t* block_info = block_info_t::at(block.address);
56 if (size_cast(block_info->subpool_index) >= m_subpools.size()) {
57 // the header of the block maybe overwritten by a reuse of block
58 continue;
59 }
60 subpool_t& owner = m_subpools[size_cast(block_info->subpool_index)];
61 if (block.n_bytes == owner.size) { // whole subpool is freed
62 // To ensure the subpool index in all blocks are valid
63 // do not erase the released subpool
64 debug_log("[gc]: subpool %zu released (%zu bytes)\n", size_cast(block_info->subpool_index) + 1, owner.size);
65 memory_freed += owner.size;
66 // release the subpool
67 q_free(pointer::remove_const(owner.begin)); // TODO: reuse the memory
68 // update total pool size
69 this->m_pool_total -= owner.size;
70 // reset the subpool to ZEROs/NULLs
71 owner = {};
72 // remove the freed block from the list
73 it = m_freed_blocks.erase(it);
74 }
75 else {
76 ++it;
77 }
78 }
79 return memory_freed;
80}
