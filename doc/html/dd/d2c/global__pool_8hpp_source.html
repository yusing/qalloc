35template <size_type POOL_SIZE>
36inline pool_pointer initialize_pool_if_needed(std::atomic<pool_pointer>& pool_atomic) {
37 static std::mutex g_pool_mutex;
38 pool_pointer p_pool = pool_atomic.load(std::memory_order_relaxed);
39 std::atomic_thread_fence(std::memory_order_acquire);
40 if (p_pool == nullptr) {
41 std::lock_guard<std::mutex> lock_guard(g_pool_mutex);
42 p_pool = pool_atomic.load(std::memory_order_relaxed);
43 if (p_pool == nullptr) { // double-check
44 p_pool = new pool_t(POOL_SIZE);
45 atomic_thread_fence(std::memory_order_release);
46 pool_atomic.store(p_pool, std::memory_order_relaxed);
47 }
48 }
49 return p_pool;
50}

53/*
54 * Using different pool for different type can reduce memory fragmentation.
55 * Might slightly improve performance.
56 */
57 thread_local std::atomic<pool_pointer> g_pool_shared;

59 template <typename T>
60 thread_local std::atomic<pool_pointer> g_pool_unique;

62 template <typename T>
63 pool_pointer get_instance() {
64#if QALLOC_DEBUG
65 // use shared pool for debug mode
66 return initialize_pool_if_needed<256>(g_pool_shared);
67#else // QALLOC_DEBUG
68 return initialize_pool_if_needed<sizeof(T) * 16>(g_pool_unique<T>); // the initial size does not affect performance much, just keep it small
69#endif // QALLOC_DEBUG
70 }
71#else // QALLOC_CXX_14
72 // use ahared pool for all types in C++11 or earlier, since variable templates is not supported
