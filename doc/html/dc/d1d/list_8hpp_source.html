41 struct node_type : public T {
42 // don't use value_type here
43 // because when T::value_type exists
44 // it maybe used instead of T
45 node_type* next;
46 explicit node_type(T&& lv) : T(std::forward<T>(lv)),
47 next(nullptr) {}
48 static node_type* set(void_ptr_t p_this, T&& lv) {
49 return new (p_this) node_type(std::forward<T>(lv));
50 }
51 };
52 using node_pointer = node_type*;

54 using iterator = class list_iterator : public qalloc::internal::iterator<node_type> {

57 using qalloc::internal::iterator<node_type>::iterator;

62 }
63 list_iterator operator++(int) {

76 inline list(size_type, value_type&& first) : m_size(1) {
77 void_ptr_t ptr = q_malloc(sizeof(node_type));
78 node_pointer node = node_type::set(ptr, std::forward<value_type>(first));
79 m_head = node;

86 inline ~list() {

88 auto* next = m_head->next;
89 m_head->~node_type();
90 q_free(m_head);
91 m_head = next;
92 }

94 template <typename ...Args>
95 void emplace_back(Args&& ...args) {
96 void_ptr_t ptr = q_malloc(sizeof(node_type));
97 node_pointer node = node_type::set(ptr, std::forward<Args>(args)...);
98 m_tail->next = node;
99 m_tail = node;
100 ++m_size;
101 }

110 QALLOC_NODISCARD constexpr bool empty() const noexcept {

112 }
