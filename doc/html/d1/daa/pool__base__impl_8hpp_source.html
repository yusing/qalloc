31inline pool_base_t::pool_base_t(size_type byte_size)
32 : m_subpools (1_z, new_subpool(byte_size)),
33 m_cur_subpool (&m_subpools.front()),

51inline subpool_t pool_base_t::new_subpool(size_type n_bytes) {
52 QALLOC_RESTRICT byte_pointer begin = static_cast<byte_pointer>(q_malloc(n_bytes)); // NOLINT(modernize-use-auto)
53 QALLOC_RESTRICT byte_pointer end = begin + n_bytes;
54 return subpool_t{
55 begin, // .begin
56 end, // .end
57 begin, // .current
58 n_bytes // .size
59 };

62inline byte_pointer pool_base_t::allocate(size_type n_bytes) const {
63 QALLOC_ASSERT(n_bytes > 0);
64 // if current pool cannot allocate n_bytes
65 // no need to check the freed blocks (assumed they are smaller than n_bytes)
66 if (can_allocate(n_bytes)) {
67 // try to find a q_free block in the freed blocks list
68 auto it = std::find_if(
69 m_freed_blocks.begin(),
70 m_freed_blocks.end(),
71 [n_bytes](const freed_block_t& block) { // find a block with enough space
72 return block.n_bytes >= n_bytes;
73 }
74 );
75 if (it != m_freed_blocks.end()) { // found a block with enough space
76 QALLOC_ASSERT(it->n_bytes != 0);
77 QALLOC_ASSERT(it->address != nullptr);
78 freed_block_t reused_block = *it;
79 m_freed_blocks.erase(it);
80 if (reused_block.n_bytes > n_bytes) { // split if it has extra space left
81 // deallocate the first (reused_block.n_bytes - n_bytes) bytes
82 size_type size_left = reused_block.n_bytes - n_bytes;
83 deallocate(reused_block.address, size_left);
84 // and then reuse from the last n_bytes bytes of the block
85 // to keep block info (i.e. subpool index and type info) at the beginning of the block
86 reused_block.address += size_left;
87 }
88 debug_log("[allocate] reused freed block of %zu bytes and has %zu bytes left @ %p (Thread %zu Subpool %zu)\n",
89 reused_block.n_bytes, reused_block.n_bytes - n_bytes, reused_block.address, thread_id(),
90 m_subpools.size());
91 QALLOC_ASSERT(reused_block.address != nullptr);
92 return reused_block.address;
93 }
94 }
95 else {
96 // memory exhausted in pool
97 // add new subpool with 2 * (n_bytes || m_cur_subpool->size) (larger one)
98 add_subpool(std::max(n_bytes * 2, m_cur_subpool->size * 2));
99 }
100
101 byte_pointer address = pointer::launder(m_cur_subpool->pos);
102 // move the current pointer
103 m_cur_subpool->pos += n_bytes;
104 QALLOC_ASSERT(address != nullptr);
105
106 debug_log("[allocate] allocated %zu bytes @ %p (Thread %zu Subpool %zu)\n", n_bytes, address, thread_id(),
107 m_subpools.size());
108 return address;
109}

111template <bool merge>
112inline void pool_base_t::deallocate(byte_pointer p, size_type n_bytes) const {
113 QALLOC_ASSERT(p != nullptr);
114 QALLOC_ASSERT(n_bytes > 0);
115 QALLOC_ASSERT(is_valid(p));
116
117 if (m_freed_blocks.empty()) {
118 m_freed_blocks.emplace_back(freed_block_t{n_bytes, p});
119 }
120 else {
121 freed_block_t freed_block{n_bytes, p};
122 // make sure the freed block is sorted by address in ascending order (in order to merge blocks)
123 auto insert_pos = std::lower_bound(m_freed_blocks.begin(), m_freed_blocks.end(), freed_block, freed_block_t::less);
124 QALLOC_IF_CONSTEXPR(merge) {
125 if (insert_pos != m_freed_blocks.end() && freed_block.is_adjacent_to(*insert_pos)) { // is adjacent to the next block
126 debug_log("[deallocate] merged %p (%zu bytes) and %p (%zu bytes) into %zu bytes (Thread %zu)\n", p, n_bytes,
127 insert_pos->address, insert_pos->n_bytes, insert_pos->n_bytes + n_bytes, thread_id());
128 // merge with the next block
129 insert_pos->n_bytes += n_bytes;
130 insert_pos->address = p;
131
132 if (m_freed_blocks.size() > 1_z) {
133 // iterate from the second block
134 insert_pos = m_freed_blocks.begin() + 1;
135 while (insert_pos != m_freed_blocks.end()) { // is adjacent to the previous block
136 auto prev = insert_pos - 1;
137 if (prev->is_adjacent_to(*insert_pos)) {
138 debug_log("[deallocate] merged %p (%zu bytes) and %p (%zu bytes) into %zu bytes (Thread %zu)\n",
139 prev->address, prev->n_bytes, insert_pos->address, insert_pos->n_bytes,
140 prev->n_bytes + insert_pos->n_bytes, thread_id());
141 // merge with the previous block
142 prev->n_bytes += insert_pos->n_bytes;
143 insert_pos = m_freed_blocks.erase(insert_pos);
144 }
145 else {
146 ++insert_pos;
147 }
148 }
149 }
150 return;
151 }
152 }
153 // no block to merge with, insert the freed block
154 debug_log("[deallocate] deallocated %zu bytes @ %p (Thread %zu Subpool %zu)\n", n_bytes, p, thread_id(),
155 m_subpools.size());
156 m_freed_blocks.emplace(insert_pos, freed_block);
157 }
158}

160inline void pool_base_t::add_subpool(size_type n_bytes) const {
161 debug_log("[allocate] adding new subpool with size %zu (Thread %zu Subpool %zu)\n", n_bytes, thread_id(),
162 m_subpools.size());
163 // if there is space left in current subpool
164 if (m_cur_subpool->end != m_cur_subpool->pos) {
165 debug_log("[allocate] subpool %zu has %zu bytes left @ %p (Thread %zu)\n", m_subpools.size(),
166 m_cur_subpool->end - m_cur_subpool->pos, m_cur_subpool->pos, thread_id());
167 // mark it as freed
168 deallocate<false>(m_cur_subpool->pos, size_cast(m_cur_subpool->end - m_cur_subpool->pos));
169 }
170 // add new subpool
171 m_subpools.emplace_back(new_subpool(n_bytes));
172 m_cur_subpool = &m_subpools.back();
173 m_pool_total += n_bytes;
174}

176constexpr bool pool_base_t::can_allocate(size_type n_bytes) const noexcept {
177 return m_cur_subpool->pos + n_bytes <= m_cur_subpool->end;
178}

180size_type pool_base_t::bytes_used() const noexcept {
181 size_t bytes_used = m_pool_total;
182 for (const auto& block : m_freed_blocks) {
183 bytes_used -= block.n_bytes;
184 }
185 bytes_used -= size_cast(m_cur_subpool->end - m_cur_subpool->pos);
186 return bytes_used;
187}

189constexpr size_type pool_base_t::pool_size() const noexcept {
190 return m_pool_total;
191}

193bool pool_base_t::is_valid(void_pointer p) const noexcept {
194 return std::any_of(
195 m_subpools.begin(),
196 m_subpools.end(),
197 [p](const subpool_t& subpool) {
198 return pointer::in_range(p, subpool.begin, subpool.end);
199 }
200 );
201}

213inline void pool_base_t::print_info(bool usage_only) const {
214 QALLOC_PRINTF("Memory Pool:");
215 size_type bytes_used = this->bytes_used();
216 QALLOC_PRINTF(" Usage: %zu of %zu bytes", bytes_used, pool_size());
217 if (pool_size() != 0_z) {
218 QALLOC_PRINTF(" (%zu%%)\n", bytes_used * 100 / pool_size());
219 }
220 else {
221 QALLOC_PRINTF("\n");
222 }
223 if (usage_only) {
224 return;
225 }
226 QALLOC_PRINTF(" Subpools: \n");
227 int i = 1;
228 for (const auto& pool : m_subpools) {
229 if (pool.begin == nullptr) {
230 QALLOC_PRINTF(" %d: released by gc\n", i);
231 }
232 else {
233 QALLOC_PRINTF(" %d: %p ~ %p (%zu bytes)\n", i, pool.begin, pool.end, pool.size);
234 QALLOC_PRINTF(" Position @ %p\n", pool.pos);
235 }
236 ++i;
237 }
238 auto& last_subpool = m_subpools.back();
239 if (last_subpool.pos != last_subpool.end) {
240 QALLOC_PRINTF(" %zu bytes unused\n", size_cast(last_subpool.end - last_subpool.pos));
241 }
242 QALLOC_PRINTF("\n Deallocated blocks:\n");
243 for (const auto& block : m_freed_blocks) {
244 auto allocated_block = block_info_t::at(block.address);
245 QALLOC_PRINTF(" %p: %zu bytes\n", block.address, block.n_bytes);
246 if (size_cast(allocated_block->subpool_index) <= m_subpools.size()) {
247 QALLOC_PRINTF(" Subpool: %zu, type: ", size_cast(allocated_block->subpool_index) + 1);
248 if (allocated_block->is_valid()) {
249 auto type_name = demangled_type_name_of(allocated_block->type_info->name());
250 QALLOC_PRINTF("%s\n", type_name.c_str());
251 }
252 else {
253 QALLOC_PRINTF("N/A\n");
254 }
255 }
256 else {
257 QALLOC_PRINTF(" Subpool: N/A, type: N/A\n");
258 }
259 }
260 QALLOC_PRINTF("\n");
261}
